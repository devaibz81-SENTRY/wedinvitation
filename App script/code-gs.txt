/**
 * Wedding Songs & Messages - Google Apps Script
 * Handles: GET ?sheet=Songs|Messages | POST ?sheet=Songs|Messages
 */

function doGet(e) {
  try {
    const params = (e && e.parameter) ? e.parameter : {};
    const sheetName = params.sheet || 'Songs';
    return jsonOut({ data: getSheetData(sheetName), sheet: sheetName });
  } catch (err) {
    return jsonOut({ error: err.toString() });
  }
}

function doPost(e) {
  try {
    const params = (e && e.parameter) ? e.parameter : {};
    const sheetName = params.sheet || 'Songs';
    
    let payload;
    if (e.postData && e.postData.contents) {
      payload = JSON.parse(e.postData.contents);
    } else {
      return jsonOut({ error: 'No data received' });
    }
    
    const sheet = getOrCreateSheet(sheetName, ['Timestamp', 'Name', 'Song', 'Date']);
    sheet.appendRow([
      new Date(),
      payload.name || 'Guest',
      payload.content || '',
      new Date().toLocaleDateString()
    ]);
    
    return jsonOut({ success: true, sheet: sheetName });
  } catch (err) {
    return jsonOut({ error: err.toString() });
  }
}

function getSheetData(sheetName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) return [];
  
  const rows = sheet.getDataRange().getValues();
  const data = [];
  
  for (let i = 1; i < rows.length; i++) {
    const row = rows[i];
    if (row[1]) {
      data.push({
        timestamp: row[0] ? row[0].toString() : '',
        name: row[1] ? row[1].toString() : '',
        content: row[2] ? row[2].toString() : '',
        date: row[3] ? row[3].toString() : ''
      });
    }
  }
  
  return data;
}

function getOrCreateSheet(name, cols) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.appendRow(cols);
    sheet.getRange(1, 1, 1, cols.length).setFontWeight('bold').setBackground('#f3f3f3');
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function jsonOut(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
