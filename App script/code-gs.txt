/**
 * Wedding Songs & Messages - Google Apps Script
 * Handles: 
 *   GET ?sheet=Songs|Messages|CheckIns - Returns JSON data
 *   GET ?action=getCheckIns - Returns check-ins list
 *   GET ?action=test - Test if script is working
 *   GET ?page=songs - Returns HTML song list page
 *   GET ?page=messages - Returns HTML interactive messages page
 *   POST ?action=checkin - Records guest check-in
 *   POST ?sheet=Songs|Messages - Adds entry to sheet
 */

function doGet(e) {
  try {
    const params = (e && e.parameter) ? e.parameter : {};
    
    // Handle HTML page requests
    if (params.page === 'songs') {
      return HtmlService.createHtmlOutputFromFile('SongsPage')
        .setTitle('Song Requests - R&A Wedding')
        .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
    }
    
    if (params.page === 'messages') {
      return HtmlService.createHtmlOutputFromFile('MessagesPage')
        .setTitle('Guest Messages - R&A Wedding')
        .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
    }
    
    // Get check-ins
    if (params.action === 'getCheckIns') {
      const checkins = getSheetData('CheckIns');
      return jsonOut({ checkins: checkins });
    }
    
    // Test endpoint
    if (params.action === 'test') {
      return jsonOut({ status: 'ok', message: 'Apps Script is working!' });
    }
    
    // Default: Return JSON data
    const sheetName = params.sheet || 'Songs';
    return jsonOut({ data: getSheetData(sheetName), sheet: sheetName });
  } catch (err) {
    return jsonOut({ error: err.toString() });
  }
}

function doPost(e) {
  try {
    const params = (e && e.parameter) ? e.parameter : {};
    
    let payload;
    if (e.postData && e.postData.contents) {
      payload = JSON.parse(e.postData.contents);
    } else {
      return jsonOut({ error: 'No data received' });
    }
    
    // Handle check-in
    if (params.action === 'checkin' || payload.action === 'checkin') {
      const sheet = getOrCreateSheet('CheckIns', ['Timestamp', 'Guest ID', 'Guest Name', 'Party Count', 'Checked In Time']);
      
      // Check if already checked in
      const existing = sheet.getDataRange().getValues();
      for (let i = 1; i < existing.length; i++) {
        if (existing[i][1] === payload.guestId) {
          return jsonOut({ success: true, message: 'Already checked in', alreadyCheckedIn: true });
        }
      }
      
      sheet.appendRow([
        new Date(),
        payload.guestId || '',
        payload.guestName || 'Guest',
        payload.partyCount || 1,
        new Date().toLocaleTimeString()
      ]);
      
      return jsonOut({ success: true, message: 'Checked in successfully' });
    }
    
    // Default: Add to Songs or Messages sheet
    const sheetName = params.sheet || 'Songs';
    const sheet = getOrCreateSheet(sheetName, ['Timestamp', 'Name', 'Song', 'Date']);
    sheet.appendRow([
      new Date(),
      payload.name || 'Guest',
      payload.content || '',
      new Date().toLocaleDateString()
    ]);
    
    return jsonOut({ success: true, sheet: sheetName });
  } catch (err) {
    return jsonOut({ error: err.toString() });
  }
}

function getSheetData(sheetName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) return [];
  
  const rows = sheet.getDataRange().getValues();
  const data = [];
  
  for (let i = 1; i < rows.length; i++) {
    const row = rows[i];
    if (row[1]) {
      data.push({
        timestamp: row[0] ? row[0].toString() : '',
        name: row[1] ? row[1].toString() : '',
        content: row[2] ? row[2].toString() : '',
        date: row[3] ? row[3].toString() : '',
        // For check-ins
        guestId: row[1] ? row[1].toString() : '',
        guestName: row[2] ? row[2].toString() : '',
        partyCount: row[3] ? row[3].toString() : '',
        checkedInTime: row[4] ? row[4].toString() : ''
      });
    }
  }
  
  return data;
}

function getOrCreateSheet(name, cols) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.appendRow(cols);
    sheet.getRange(1, 1, 1, cols.length).setFontWeight('bold').setBackground('#f3f3f3');
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function jsonOut(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
