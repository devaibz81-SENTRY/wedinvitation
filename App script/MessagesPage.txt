  <script>
    var messages = [];
    var currentIndex = 0;
    
    function loadMessages() {
      google.script.run
        .withSuccessHandler(function(data) {
          messages = data || [];
          render();
        })
        .getSheetData('Messages');
    }
    
    function render() {
      const card = document.getElementById('card');
      const empty = document.getElementById('empty');
      const messageEl = document.getElementById('message');
      const authorEl = document.getElementById('author');
      const progressEl = document.getElementById('progress');
      const prevBtn = document.getElementById('prevBtn');
      const nextBtn = document.getElementById('nextBtn');
      const dotsEl = document.getElementById('dots');
      
      if (messages.length === 0) {
        card.style.display = 'none';
        document.querySelector('.controls').style.display = 'none';
        document.querySelector('.dots').style.display = 'none';
        empty.style.display = 'block';
        document.getElementById('progress').textContent = '0 messages';
        return;
      }
      
      empty.style.display = 'none';
      card.style.display = 'flex';
      document.querySelector('.controls').style.display = 'flex';
      dotsEl.style.display = 'flex';
      
      updateDisplay();
    }
    
    function updateDisplay() {
      const messageEl = document.getElementById('message');
      const authorEl = document.getElementById('author');
      const progressEl = document.getElementById('progress');
      const prevBtn = document.getElementById('prevBtn');
      const nextBtn = document.getElementById('nextBtn');
      const dotsEl = document.getElementById('dots');
      
      const msg = messages[currentIndex];
      messageEl.textContent = '"' + msg.content + '"';
      authorEl.textContent = '— ' + msg.name;
      progressEl.textContent = (currentIndex + 1) + ' of ' + messages.length;
      
      prevBtn.disabled = currentIndex === 0;
      nextBtn.disabled = currentIndex === messages.length - 1;
      
      dotsEl.innerHTML = messages.map(function(_, i) {
        return '<div class="dot' + (i === currentIndex ? ' active' : '') + '"></div>';
      }).join('');
    }
    
    function prevMessage() {
      if (currentIndex > 0) {
        currentIndex--;
        updateDisplay();
      }
    }
    
    function nextMessage() {
      if (currentIndex < messages.length - 1) {
        currentIndex++;
        updateDisplay();
      }
    }
    
    document.addEventListener('keydown', function(e) {
      if (e.key === 'ArrowLeft') prevMessage();
      if (e.key === 'ArrowRight') nextMessage();
    });
    
    loadMessages();
  </script>
